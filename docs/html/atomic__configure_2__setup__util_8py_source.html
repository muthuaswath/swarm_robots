69def rollback_env_variables(environ, env_var_subfolders):
70 """
71 Generate shell code to reset environment variables.
72
73 by unrolling modifications based on all workspaces in CMAKE_PREFIX_PATH.
74 This does not cover modifications performed by environment hooks.
75 """
76 lines = []
77 unmodified_environ = copy.copy(environ)
78 for key in sorted(env_var_subfolders.keys()):
79 subfolders = env_var_subfolders[key]
80 if not isinstance(subfolders, list):
81 subfolders = [subfolders]
82 value = _rollback_env_variable(unmodified_environ, key, subfolders)
83 if value is not None:
84 environ[key] = value
85 lines.append(assignment(key, value))
86 if lines:
87 lines.insert(0, comment('reset environment variables by unrolling modifications based on all workspaces in CMAKE_PREFIX_PATH'))
88 return lines

91def _rollback_env_variable(environ, name, subfolders):
92 """
93 For each catkin workspace in CMAKE_PREFIX_PATH remove the first entry from env[NAME] matching workspace + subfolder.
94
95 :param subfolders: list of str '' or subfoldername that may start with '/'
96 :returns: the updated value of the environment variable.
97 """
98 value = environ[name] if name in environ else ''
99 env_paths = [path for path in value.split(os.pathsep) if path]
100 value_modified = False
101 for subfolder in subfolders:
102 if subfolder:
103 if subfolder.startswith(os.path.sep) or (os.path.altsep and subfolder.startswith(os.path.altsep)):
104 subfolder = subfolder[1:]
105 if subfolder.endswith(os.path.sep) or (os.path.altsep and subfolder.endswith(os.path.altsep)):
106 subfolder = subfolder[:-1]
107 for ws_path in _get_workspaces(environ, include_fuerte=True, include_non_existing=True):
108 path_to_find = os.path.join(ws_path, subfolder) if subfolder else ws_path
109 path_to_remove = None
110 for env_path in env_paths:
111 env_path_clean = env_path[:-1] if env_path and env_path[-1] in [os.path.sep, os.path.altsep] else env_path
112 if env_path_clean == path_to_find:
113 path_to_remove = env_path
114 break
115 if path_to_remove:
116 env_paths.remove(path_to_remove)
117 value_modified = True
118 new_value = os.pathsep.join(env_paths)
119 return new_value if value_modified else None

122def _get_workspaces(environ, include_fuerte=False, include_non_existing=False):
123 """
124 Based on CMAKE_PREFIX_PATH return all catkin workspaces.
125
126 :param include_fuerte: The flag if paths starting with '/opt/ros/fuerte' should be considered workspaces, ``bool``
127 """
128 # get all cmake prefix paths
129 env_name = 'CMAKE_PREFIX_PATH'
130 value = environ[env_name] if env_name in environ else ''
131 paths = [path for path in value.split(os.pathsep) if path]
132 # remove non-workspace paths
133 workspaces = [path for path in paths if os.path.isfile(os.path.join(path, CATKIN_MARKER_FILE)) or (include_fuerte and path.startswith('/opt/ros/fuerte')) or (include_non_existing and not os.path.exists(path))]
134 return workspaces

137def prepend_env_variables(environ, env_var_subfolders, workspaces):
138 """Generate shell code to prepend environment variables for the all workspaces."""
139 lines = []
140 lines.append(comment('prepend folders of workspaces to environment variables'))
141
142 paths = [path for path in workspaces.split(os.pathsep) if path]
143
144 prefix = _prefix_env_variable(environ, 'CMAKE_PREFIX_PATH', paths, '')
145 lines.append(prepend(environ, 'CMAKE_PREFIX_PATH', prefix))
146
147 for key in sorted(key for key in env_var_subfolders.keys() if key != 'CMAKE_PREFIX_PATH'):
148 subfolder = env_var_subfolders[key]
149 prefix = _prefix_env_variable(environ, key, paths, subfolder)
150 lines.append(prepend(environ, key, prefix))
151 return lines

154def _prefix_env_variable(environ, name, paths, subfolders):
155 """
156 Return the prefix to prepend to the environment variable NAME.
157
158 Adding any path in NEW_PATHS_STR without creating duplicate or empty items.
159 """
160 value = environ[name] if name in environ else ''
161 environ_paths = [path for path in value.split(os.pathsep) if path]
162 checked_paths = []
163 for path in paths:
164 if not isinstance(subfolders, list):
165 subfolders = [subfolders]
166 for subfolder in subfolders:
167 path_tmp = path
168 if subfolder:
169 path_tmp = os.path.join(path_tmp, subfolder)
170 # skip nonexistent paths
171 if not os.path.exists(path_tmp):
172 continue
173 # exclude any path already in env and any path we already added
174 if path_tmp not in environ_paths and path_tmp not in checked_paths:
175 checked_paths.append(path_tmp)
176 prefix_str = os.pathsep.join(checked_paths)
177 if prefix_str != '' and environ_paths:
178 prefix_str += os.pathsep
179 return prefix_str

196def prepend(environ, key, prefix):
197 if key not in environ or not environ[key]:
198 return assignment(key, prefix)
199 if not IS_WINDOWS:
200 return 'export %s="%s$%s"' % (key, prefix, key)
201 else:
202 return 'set %s=%s%%%s%%' % (key, prefix, key)

205def find_env_hooks(environ, cmake_prefix_path):
206 """Generate shell code with found environment hooks for the all workspaces."""
207 lines = []
208 lines.append(comment('found environment hooks in workspaces'))
209
210 generic_env_hooks = []
211 generic_env_hooks_workspace = []
212 specific_env_hooks = []
213 specific_env_hooks_workspace = []
214 generic_env_hooks_by_filename = {}
215 specific_env_hooks_by_filename = {}
216 generic_env_hook_ext = 'bat' if IS_WINDOWS else 'sh'
217 specific_env_hook_ext = environ['CATKIN_SHELL'] if not IS_WINDOWS and 'CATKIN_SHELL' in environ and environ['CATKIN_SHELL'] else None
218 # remove non-workspace paths
219 workspaces = [path for path in cmake_prefix_path.split(os.pathsep) if path and os.path.isfile(os.path.join(path, CATKIN_MARKER_FILE))]
220 for workspace in reversed(workspaces):
221 env_hook_dir = os.path.join(workspace, 'etc', 'catkin', 'profile.d')
222 if os.path.isdir(env_hook_dir):
223 for filename in sorted(os.listdir(env_hook_dir)):
224 if filename.endswith('.%s' % generic_env_hook_ext):
225 # remove previous env hook with same name if present
226 if filename in generic_env_hooks_by_filename:
227 i = generic_env_hooks.index(generic_env_hooks_by_filename[filename])
228 generic_env_hooks.pop(i)
229 generic_env_hooks_workspace.pop(i)
230 # append env hook
231 generic_env_hooks.append(os.path.join(env_hook_dir, filename))
232 generic_env_hooks_workspace.append(workspace)
233 generic_env_hooks_by_filename[filename] = generic_env_hooks[-1]
234 elif specific_env_hook_ext is not None and filename.endswith('.%s' % specific_env_hook_ext):
235 # remove previous env hook with same name if present
236 if filename in specific_env_hooks_by_filename:
237 i = specific_env_hooks.index(specific_env_hooks_by_filename[filename])
238 specific_env_hooks.pop(i)
239 specific_env_hooks_workspace.pop(i)
240 # append env hook
241 specific_env_hooks.append(os.path.join(env_hook_dir, filename))
242 specific_env_hooks_workspace.append(workspace)
243 specific_env_hooks_by_filename[filename] = specific_env_hooks[-1]
244 env_hooks = generic_env_hooks + specific_env_hooks
245 env_hooks_workspace = generic_env_hooks_workspace + specific_env_hooks_workspace
246 count = len(env_hooks)
247 lines.append(assignment('_CATKIN_ENVIRONMENT_HOOKS_COUNT', count))
248 for i in range(count):
249 lines.append(assignment('_CATKIN_ENVIRONMENT_HOOKS_%d' % i, env_hooks[i]))
250 lines.append(assignment('_CATKIN_ENVIRONMENT_HOOKS_%d_WORKSPACE' % i, env_hooks_workspace[i]))
251 return lines
